<script>
    const GIZLI_ANAHTAR = "LegologlarFLL2024";
    const DEPO_URL = "https://script.google.com/macros/s/AKfycbxww2-ThoWar93P94sxsl_3JxHAY7_RrXjYEfM2i0FVn2LLBozDkkTu6MiU-wCrEQwLtg/exec";
    const BACKEND_URL = "https://script.google.com/macros/s/AKfycbyFcIai4hAuiVG_M3q_vYnEQS5R1LFq14izib1MqoO2vXOe4Zi2IHR5mYuSOlPYEpye/exec";

    let base64Image = "";
    var map = L.map('map').setView([39.0, 35.0], 6);
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);
    var marker = null, selectedLoc = null;

    map.on('click', e => {
        selectedLoc = e.latlng;
        if(marker) map.removeLayer(marker);
        marker = L.marker(selectedLoc).addTo(map);
    });

    // FotoÄŸrafÄ± Base64'e Ã§evirme
    function handleFile(input) {
        const file = input.files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = e => {
                base64Image = e.target.result;
                const img = document.getElementById('imagePreview');
                img.src = base64Image;
                img.style.display = "block";
                document.getElementById('upload-text').style.display = "none";
            };
            reader.readAsDataURL(file);
        }
    }

    function sifrele(metin) { return CryptoJS.AES.encrypt(metin, GIZLI_ANAHTAR).toString(); }
    function sifreCoz(sifreliMetin) {
        try { return CryptoJS.AES.decrypt(sifreliMetin, GIZLI_ANAHTAR).toString(CryptoJS.enc.Utf8); } catch(e) { return null; }
    }

    async function submitReport() {
        const desc = document.getElementById('rDesc').value;
        if(!selectedLoc || desc.length < 5) { alert("Konum ve aÃ§Ä±klama eksik!"); return; }

        const rawData = { 
            detay: desc, 
            foto: base64Image, 
            koordinat: selectedLoc, 
            tarih: new Date().toLocaleString() 
        };

        try {
            const secureData = sifrele(JSON.stringify(rawData));
            await fetch(DEPO_URL, { method: 'POST', mode: 'no-cors', body: JSON.stringify({ secureData }) });
            await fetch(BACKEND_URL, { method: 'POST', mode: 'no-cors' });
            alert("Ä°hbar ve fotoÄŸraf merkeze iletildi!");
            location.reload();
        } catch(e) { alert("Hata!"); }
    }

    // --- ADMIN PANELÄ° (FOTOÄRAF GÃ–RÃœNÃœMÃœ EKLENDÄ°) ---
    async function checkAdmin() {
        if (new URLSearchParams(window.location.search).get('admin') === 'true') {
            if (prompt("Åifre:") === "1234") renderAdmin();
        }
    }

    async function renderAdmin() {
        document.body.innerHTML = `<div class="section" style="max-width:98%">
            <h2>ğŸ•µï¸ YÃ¶netici Paneli</h2>
            <div id="admin-list">YÃ¼kleniyor...</div>
        </div>`;
        const res = await fetch(DEPO_URL);
        const data = await res.json();
        let html = "<table border='1' style='width:100%; border-collapse:collapse;'><tr><th>Tarih</th><th>Detay</th><th>FotoÄŸraf</th><th>Konum</th></tr>";
        
        data.forEach(row => {
            const item = JSON.parse(sifreCoz(row[1]));
            if(item) {
                const imgTag = item.foto ? `<img src="${item.foto}" style="width:80px; height:80px; object-fit:cover; cursor:pointer" onclick="window.open('${item.foto}')">` : "Yok";
                html += `<tr>
                    <td style="padding:10px">${item.tarih}</td>
                    <td style="padding:10px">${item.detay}</td>
                    <td style="padding:10px; text-align:center">${imgTag}</td>
                    <td style="padding:10px">${item.koordinat.lat.toFixed(4)}, ${item.koordinat.lng.toFixed(4)}</td>
                </tr>`;
            }
        });
        document.getElementById('admin-list').innerHTML = html + "</table>";
    }

    window.onload = () => {
        checkAdmin();
        fetch(BACKEND_URL).then(r => r.json()).then(d => document.getElementById('countVal').innerText = d.total);
    };
